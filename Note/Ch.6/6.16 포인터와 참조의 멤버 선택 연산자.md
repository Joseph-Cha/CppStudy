``` C++
#include <iostream>
#include <cstddef>

using namespace std;

struct Person
{
    int age;
    double weight;
};

int main()
{
    // 보통
    Person person;
    person.age = 5;
    person.weight = 30;

    // 참조일 때는 .를 쓴다.
    Person &ref = person;
    ref.age = 15;
    
    // 포인터일 때는 ->를 쓴다
    Person *ptr = &person;
    ptr->age = 30;
    // (*ptr).age = 20;

    Person &ref2 = *ptr;
    ref2.age = 45;
    
    cout << &person << endl;    // 0x62fdf0
    cout << &ref2 << endl;      // 0x62fdf0

    return 0; 
}
```
