#include <iostream>

    const int value = 5;

    // ptr이 집고 있는 value의 값 const로 집고 있기 때문에 write는 x

    *ptr = 6; // error

    // 주소의 값을 바꾸지 않겠다는 것은 아니다
    // const ptr에 새로운 주소를 할당 할 수는 있다.
    int value2 = 5;
    const int *ptr = &value2;

    // 포인터 자체가 상수로 바꿈
